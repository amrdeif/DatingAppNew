=========Angular=========

--Install angilar version
npm install -g @angular/cli

--Install bootstrap and font awesome
npm install bootstrap font-awesome

--Gets npm version
npm --version

--Create new angular app
ng new DatingApp-SPA

--Run application
ng serve



=======Dot Net Core=======

--Create a new API
dotnet new webapi -n DatingApp.API

--Run application
dotnet run

--Run application and listen to changes
dotnet watch run
